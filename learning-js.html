// The 'while' funtion loops until we reach 100.

// Create a <p> element and store it in 'newP' variable.

var newP = document.createElement('p');

// Create a node of text, store it in a variable.

var num2Text = document.createTextNode(num2);

// Add the new text, using the variable, to the <p>.

newP.appendChild(num2Text);

// Locate where to add our new <p> elements and store in a variable.

var outputBox = document.getElementById("externalScriptOutput");

// Add the new <p> elements to the page within the parent element that we found earlier.

outputBox.appendChild(newP);

// Add 10 to the number.

num2 += 10;

}
